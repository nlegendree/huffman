\begin{algorithme}
\begin{enregistrement}{Octet}{
    \champEnregistrement{octet}{Naturel8Bits}
}
\end{enregistrement}

\fonction{octet}
    {nombre : Naturel8Bits}{Octet}
    {}{sortie : Octet}
{
    \affecter{sortie.octet}{nombre}
    \retourner{sortie}
}

\fonction{obtenirIemeBit}
    {octet : Octet, pos : \naturelNonNul}{bit: Bit}
    {pos $\leq$ 8}{estUn : \booleen}
{
    \affecter{estUn}{(\champ{octet}{octet} $>>$ i - 1) \& 1}
    \retourner{bit(estUn)}
}

\procedure{fixerIemeBit}{
    \paramEntreeSortie{octet : Octet}
    \paramEntree{pos : \naturelNonNul, bit: Bit}}
    {pos $\leq$ 8}{}
{
    \affecter {octet.octet}{(octet \& $~$(1 $<<$ pos) $|$ (bitEnEntier(bit) $<<$ pos))} 
}
 
\fonction{octetEnNaturel}
    {octet : Octet}{\naturel}
    {}{}
{   
    \retourner{\champ{octet}{octet}}
}

\end{algorithme}
