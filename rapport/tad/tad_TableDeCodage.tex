\begin{tad}

    \tadNom{TableDeCodage}

    \tadParametres{Element, Code} %Code = CodeBinaire

    \tadDependances{\naturel, \booleen}

    \begin{tadOperations}{TableDeCodage}
        \tadOperation{tableDeCodage}%
            {}
            {\tadUnParam{TableDeCodage}}
        \tadOperation{estVide}%
            {\tadUnParam{TableDeCodage}}%
            {\tadUnParam{\booleen}}
        \tadOperation{elementPresent}%
            {\tadDeuxParams{TableDeCodage}{Element}}%
            {\tadUnParam{\booleen}}
        \tadOperationAvecPreconditions{ajouterElement}%
            {\tadTroisParams{TableDeCodage}{Element}{Code}}%
            {\tadUnParam{TableDeCodage}}
        \tadOperationAvecPreconditions{obtenirCodeElement}%
            {\tadDeuxParams{TableDeCodage}{Element}}%
            {\tadUnParam{Code}}
        \tadOperation{codePresent}% 
            {\tadDeuxParams{TableDeCodage}{Code}}%
            {\tadUnParam{\booleen}}
        \tadOperationAvecPreconditions{obtenirElementCode}%
            {\tadDeuxParams{TableDeCodage}{Code}}%
            {\tadUnParam{Element}}
    \end{tadOperations}

    \begin{tadAxiomes}
        \tadAxiome{estVide(tableDeCodage())}
        \tadAxiome{non(estVide(ajouterElement(table, elem, code)))}
        \tadAxiome{elementPresent(ajouterElement(table, elem, code), elem)}
        \tadAxiome{elementPresent(ajouterElement(stat, elem, code), elem)}
        \tadAxiome{codePresent(ajouterElement(table, elem, code), code)}
        \tadAxiome{obtenirElementCode(ajouterElement(table, elem, code), code) = elem}
        \tadAxiome{obtenirCodeElement(ajouterElement(table, elem, code), elem) = code}
    \end{tadAxiomes}

    \begin{tadPreconditions}{}
        \tadPrecondition{ajouterElement(table, elem, code)}{non codePresent(table, code) ET non elementPresent(table, elem)}
        \tadPrecondition{obtenirCodeElement(table, elem)}{elementPresent(table, elem)}
        \tadPrecondition{obtenirElementCode(table, code)}{codePresent(table, code)}
    \end{tadPreconditions}

\end{tad}
