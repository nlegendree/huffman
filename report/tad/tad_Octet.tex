\begin{tad}

    \tadNom{Octet}

    %\tadParametres{}

    \tadDependances{\booleen, Bit, \naturel}

    \begin{tadOperations}{Octet}
        \tadOperation{octet}%
            {\tadHuitParams{Bit}{Bit}{Bit}{Bit}{Bit}{Bit}{Bit}{Bit}}%
            {\tadUnParam{Octet}}
        \tadOperation{estEgal}%
            {\tadDeuxParams{Octet}{Octet}}%
            {\tadUnParam{\booleen}}
        \tadOperation{obtenirBit}%
            {\tadDeuxParams{Octet}{1..8}}%
            {\tadUnParam{Bit}}
    \end{tadOperations}

    \begin{tadSemantiques}{Octet}
		\tadSemantique{octet}%
			{Crée un octet avec les 8 bits spécifiés.}%
        \tadSemantique{estEgal}%
			{Vérifie si les deux octets sont égaux.}  
        \tadSemantique{obtenirBit}%
			{Obtient le bit à la position spécifiée dans l'octet (position de 0 à 7).}  
    \end{tadSemantiques}

    \begin{tadAxiomes}
        \tadAxiome{obtenirBit (00000001, 1) = 1}
        \tadAxiome{estEgal (octet1,octet2) = (obtenirBit (octet1, {1..8}) = obtenirBit (octet2, {1..8}))}
    \end{tadAxiomes}

\end{tad}
