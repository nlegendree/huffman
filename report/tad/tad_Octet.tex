\begin{tad}

    \tadNom{Octet}

    %\tadParametres{}

    \tadDependances{\booleen, Bit, \naturel}

    \begin{tadOperations}{Octet}
        \tadOperation{octet}%
            {\tadHuitParams{Bit}{Bit}{Bit}{Bit}{Bit}{Bit}{Bit}{Bit}}%
            {\tadUnParam{Octet}}
        \tadOperation{estEgal}%
            {\tadDeuxParams{Octet}{Octet}}%
            {\tadUnParam{\booleen}}
        \tadOperation{obtenirIemeBit}%
            {\tadDeuxParams{Octet}{0..7}}%
            {\tadUnParam{Bit}}
        \tadOperation{fixerIemeBit}%
            {\tadTroisParams{Octet}{0..7}{Bit}}%
            {\tadUnParam{Octet}}
    \end{tadOperations}

    \begin{tadSemantiques}{Octet}
		\tadSemantique{octet}%
			{Crée un octet avec les 8 bits spécifiés.}%
        \tadSemantique{estEgal}%
			{Vérifie si les deux octets sont égaux.}  
        \tadSemantique{obtenirIemeBit}%
			{Obtient le bit à la position spécifiée dans l'octet (position de 0 à 7).}
        \tadSemantique{fixerIemeBit}%
			{Fixe un bit à la position spécifiée dans l'octet (position de 0 à 7).}    
    \end{tadSemantiques}

    \begin{tadAxiomes}
        \tadAxiome{obtenirIemeBit (00000001, 0) = 1}
        \tadAxiome{fixerIemeBit (00000001, 0, 0) = 00000000}
        \tadAxiome{estEgal (octet1,octet2) = (obtenirBit (octet1, {0..7}) = obtenirBit (octet2, {0..7}))}
    \end{tadAxiomes}

\end{tad}
