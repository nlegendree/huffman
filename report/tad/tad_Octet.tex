\begin{tad}

    \tadNom{Octet}

    \tadDependances{\booleen, Bit, \naturel}

    \begin{tadOperations}{Octet}
        \tadOperation{octet}%
            {\tadParams{}}%
            {\tadParams{Octet}}
        \tadOperationAvecPreconditions{obtenirIemeBit}%
            {\tadParams{Octet, \naturelNonNul}}%
            {\tadParams{Bit}}
        \tadOperationAvecPreconditions{fixerIemeBit}%
            {\tadParams{Octet, \naturelNonNul, Bit}}%
            {\tadParams{Octet}}
        \tadOperation{octetEnNaturel}%
            {\tadParams{Octet}}%
            {\tadParams{\naturel}}
    \end{tadOperations}

    \begin{tadPreconditions}{}
        \tadPrecondition{obtenirIemeBit(o, pos)}{pos $\geq$ 0 et pos $\leq$ 7}
        \tadPrecondition{fixerIemeBit(o, pos, bit)}{pos $\geq$ 0 et pos $\leq$ 7}
    \end{tadPreconditions}

    \begin{tadAxiomes}
        \tadAxiome{obtenirIemeBit(fixerIemeBit(octet(), i, bit), i) = bit}
    \end{tadAxiomes}

\end{tad}
