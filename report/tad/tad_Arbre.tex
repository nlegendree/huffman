\begin{tad}

    \tadNom{ArbreDeHuffman}

    \tadParametres{Element}

    \tadDependances{\naturel, \booleen}

    \begin{tadOperations}{ArbreDeHuffman}
        \tadOperation{arbreDeHuffman}%
            {\tadParams{Element, \naturel}}%
            {\tadParams{ArbreDeHuffman}}
        \tadOperation{combiner}%
            {\tadParams{ArbreDeHuffman, ArbreDeHuffman}}%
            {\tadParams{ArbreDeHuffman}}
        \tadOperation{obtenirPonderation}%
            {\tadParams{ArbreDeHuffman}}%
            {\tadParams{\naturel}}
        \tadOperationAvecPreconditions{obtenirElement}%
            {\tadParams{ArbreDeHuffman}}%
            {\tadParams{Element}}
        \tadOperation{estUneFeuille}%
            {\tadParams{ArbreDeHuffman}}%
            {\tadParams{\booleen}}
        \tadOperationAvecPreconditions{obtenirArbreGauche}%
            {\tadParams{ArbreDeHuffman}}%
            {\tadParams{ArbreDeHuffman}}
        \tadOperationAvecPreconditions{obtenirArbreDroit}%
            {\tadParams{ArbreDeHuffman}}%
            {\tadParams{ArbreDeHuffman}}
    \end{tadOperations}

    \begin{tadAxiomes}
        \tadAxiome{obtenirElement(arbreDeHuffman(element,ponderation)) = element}
        \tadAxiome{obtenirPonderation(arbreDeHuffman(element,ponderation)) = ponderation}
        \tadAxiome{obtenirPonderation(combiner(arbreGauche,arbreDroit)) = obtenirPonderation(arbreGauche) + obtenirPonderation(arbreDroit)}
        \tadAxiome{estUneFeuille(arbreDeHuffman(element,ponderation))}
        \tadAxiome{non estUneFeuille(combiner(arbreGauche,arbreDroit))}
        \tadAxiome{obtenirArbreGauche(combiner(arbreGauche,arbreDroit)) = arbreGauche}
        \tadAxiome{obtenirArbreDroit(combiner(arbreGauche,arbreDroit)) = arbreDroit}
    \end{tadAxiomes}

    \begin{tadPreconditions}{}
        \tadPrecondition{obtenirElement(arbre)}{estUneFeuille(arbre)}
        \tadPrecondition{obtenirArbreGauche(arbre)}{non estUneFeuille(arbre)}
        \tadPrecondition{obtenirArbreDroit(arbre)}{non estUneFeuille(arbre)}
    \end{tadPreconditions}

\end{tad}
