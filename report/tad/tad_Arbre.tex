\begin{tad}

    \tadNom{ArbreDeHuffman}

    \tadParametres{Element}

    \tadDependances{\naturel, \booleen}

    \begin{tadOperations}{ArbreDeHuffman}
        \tadOperation{arbreDeHuffman}%
            {\tadParams{Element, \naturel}}%
            {\tadParams{ArbreDeHuffman}}
        \tadOperation{combiner}%
            {\tadParams{ArbreDeHuffman, ArbreDeHuffman}}%
            {\tadParams{ArbreDeHuffman}}
        \tadOperation{obtenirPondération}%
            {\tadParams{ArbreDeHuffman}}%
            {\tadParams{\naturel}}
        \tadOperationAvecPreconditions{obtenirElement}%
            {\tadParams{ArbreDeHuffman}}%
            {\tadParams{Element}}
        \tadOperation{estUneFeuille}%
            {\tadParams{ArbreDeHuffman}}%
            {\tadParams{\booleen}}
        \tadOperationAvecPreconditions{obtenirArbreGauche}%
            {\tadParams{ArbreDeHuffman}}%
            {\tadParams{ArbreDeHuffman}}
        \tadOperationAvecPreconditions{obtenirArbreDroit}%
            {\tadParams{ArbreDeHuffman}}%
            {\tadParams{ArbreDeHuffman}}
    \end{tadOperations}

    \begin{tadAxiomes}
        \tadAxiome{\textsf{obtenirElement(arbreDeHuffman(élément,pondération))} = \textsf{élément}}
        \tadAxiome{\textsf{obtenirPondération(arbreDeHuffman(élément,pondération))} = \textsf{pondération}}
        \tadAxiome{\textsf{obtenirPondération(combiner(arbreGauche,arbreDroit))} = \textsf{obtenirPondération(arbreGauche)+obtenirPondération(arbreDroit)}}
        \tadAxiome{\textsf{estUneFeuille(arbreDeHuffman(élément,pondération))}}
        \tadAxiome{\textsf{non estUneFeuille(combiner(arbreGauche,arbreDroit))}}
        \tadAxiome{\textsf{obtenirArbreGauche(combiner(arbreGauche,arbreDroit))} = \textsf{arbreGauche}}
        \tadAxiome{\textsf{obtenirArbreDroit(combiner(arbreGauche,arbreDroit))} = \textsf{arbreDroit}}
    \end{tadAxiomes}

    \begin{tadPreconditions}{}
        \tadPrecondition{obtenirElement(arbre)}{estUneFeuille(arbre)}
        \tadPrecondition{obtenirArbreGauche(arbre)}{non estUneFeuille(arbre)}
        \tadPrecondition{obtenirArbreDroit(arbre)}{non estUneFeuille(arbre)}
    \end{tadPreconditions}

\end{tad}
