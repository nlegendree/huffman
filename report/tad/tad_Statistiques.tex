\begin{tad}

    \tadNom{Statistiques}

    \tadParametres{Element}

    \tadDependances{\naturel}

    \begin{tadOperations}{Statistiques}
        \tadOperation{statistiques}%
            {}
            {\tadUnParam{Statistiques}}
        \tadOperation{obtenirOccurenceElement}%
            {\tadDeuxParams{Statistiques}{Element}}%
            {\tadUnParam{\naturel}}
        \tadOperation{incrementerOccurenceElement}%
            {\tadDeuxParams{Statistiques}{Element}}%
            {\tadUnParam{Statistiques}}
        \tadOperation{fixerOccurenceElement}%
            {\tadTroisParams{Element}{\naturel}{Statistiques}}%
            {\tadUnParam{Statistiques}}
        \tadOperation{obtenirTotalOccurence}%
            {\tadUnParam{Statistiques}}%
            {\tadUnParam{\naturel}}
    \end{tadOperations}

    \begin{tadAxiomes}
        \tadAxiome{obtenirTotalOccurence(statistiques()) = 0}
        \tadAxiome{obtenirOccurenceElement(statistiques(), elem) = 0}
        \tadAxiome{obtenirOccurenceElement(incrementerOccurenceElement(stat, elem), elem) = obtenirOccurenceElement(stat, elem) + 1}
        \tadAxiome{obtenirOccurenceElement(fixerOccurenceElement(elem, nombre, stat), elem) = nombre}
        \tadAxiome{obtenirTotalOccurence(incrementerOccurenceElement(stat, elem)) = obtenirTotalOccurence(stat) + 1}
    \end{tadAxiomes}

\end{tad}
