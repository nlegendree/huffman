\begin{tad}
    \tadNom{FileDePriorité}

    \tadParametres{Element}

    \tadDependances{\naturel, \booleen}

    \begin{tadOperations}{FileDePriorité}
        \tadOperation{fileDePriorité}{}{\tadParams{FileDePriorité}}
        \tadOperation{inserer}{\tadParams{FileDePriorité, Element,  NaturelNonNul}}{\tadParams{FileDePriorité}}
        \tadOperation{supprimerPremier}{\tadParams{FileDePriorité}}{\tadParams{FileDePriorité}}
        \tadOperation{supprimerDernier}{\tadParams{FileDePriorité}}{\tadParams{FileDePriorité}}
        \tadOperation{obtenirPremier}{\tadParams{FileDePriorité}}{\tadParams{Element}}
        \tadOperation{obtenirDernier}{{FileDePriorité}}{{Element}}
        \tadOperation{estVide}{\tadParams{FileDePriorité}}{\tadParams{Boolean}}
        \tadOperation{longueur}{\tadParams{FileDePriorité}}{\tadParams{Naturel}}
    \end{tadOperations}

    \begin{tadAxiomes}
        \tadAxiome{estVide(fileDePriorite())}
        \tadAxiome{non estVide(inserer(fileDePriorite(), el, 1))}
        \tadAxiome{obtenirPremier(inserer(fileDePriorite(), el, 1)) = el}
        \tadAxiome{obtenirDernier(inserer(fileDePriorite(), el, 1)) = el}
        \tadAxiome{longueur(fileDePriorite()) = 0}
        \tadAxiome{longueur(inserer(l, e, p)) = 1 + longueur(l)}
        \tadAxiome{supprimerDernier()}
    \end{tadAxiomes}

    \begin{tadPreconditions}{}
        \tadPrecondition{supprimerPremier(l)}{non(estVide(l))}
        \tadPrecondition{supprimerDernier(l)}{non(estVide(l))}
        \tadPrecondition{obtenirPremier(l)}{non(estVide(l))}
        \tadPrecondition{obtenirDernier(l)}{non(estVide(l))}
    \end{tadPreconditions}
\end{tad}
