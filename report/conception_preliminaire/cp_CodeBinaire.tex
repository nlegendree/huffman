    {\paramEntree{codeBinaire : CodeBinaire}}

    {\paramEntreeSortie{codeBinaire : CodeBinaire}}

    {\paramEntreeSortie{codeBinaire : CodeBinaire}}
    {\paramEntree{positionIemeBit : \naturelNonNul, bitAModifier : Bit}}
    {(positionIemeBit $\leq$ longueur(codeBinaire)) et (positionIemeBit $>$ 0)}

    {codeBinaire : CodeBinaire}{\naturelNonNul}
