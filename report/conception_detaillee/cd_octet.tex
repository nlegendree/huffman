\begin{algorithme}
\begin{enregistrement}{Octet}{
    \champEnregistrement{octet}{Naturel8Bits}
}
\end{enregistrement}

\fonction{octet}
    {nombre : Naturel8Bits}{Octet}
    {}{sortie : Octet}
{
    \affecter{sortie.octet}{nombre}
    \retourner{sortie}
}

\fonction{obtenirIemeBit}
    {octet : Octet, pos : \naturelNonNul}{bit: Bit}
    {pos $>$ 0 et pos $<$ 7}{estUn : \booleen}
{
    \affecter{estUn}{(\champ{octet}{octet} \& (1 \textless \textless (i-1)))$>$ 0}
    \retourner{bit(estUn)}
}

\procedure{fixerIemeBit}{
    \paramEntree{pos : \naturelNonNul, bit: Bit}
    \paramEntreeSortie{octet : Octet}}
    {pos $>$ 0 et pos $<$ 7}{}
{
    \sialorssinon{estUn(bit)}{
        \affecter{\champ{octet}{octet}}{\champ{octet}{octet} | (1 \textless \textless (i-1))}
    }
    {
        \affecter{\champ{octet}{octet}}{\champ{octet}{octet} \& non({obtenirIemeBit(octet, i)})}
    }
}

\fonction{octetEnNaturel}
    {octet : Octet}{\naturel}
    {}{}
{   
    \retourner{\champ{octet}{octet}}
}

\end{algorithme}
