\begin{algorithme}

    \begin{enregistrement}{Statistiques}
        \champEnregistrement{taille}{\naturel}
        \champEnregistrement{element}{Element} $occurences de l'élément$
        \champEnregistrement{code}{Code}
    \end{enregistrement}

    \fonction{statistiques}
    {element : Element}{Statistiques}
    {}{}
    {b $\geq$0}{}
    {\retourner{a/b}}

    \fonction{obtenirOccurenceElement}
    {statistiques : Statistiques, element : Element}{\naturel}
    {}{}
    \pourChaque{statistiques.element}{statistiques}{
        \sialors{statistiques.element = element}
            {\retourner occurence}}
    {\retourner{0}}

    \procedure{incrementerOccurenceElement}{
        \paramEntree{element : Element}
        \paramEntreeSortie{statistiques : Statistiques}}
    {}{}
    {
    \pourChaque{statistiques.element}{statistiques}{
        \sialors{statistiques.element = element}
            {\affecter{occurence}{occurence + 1}}
        }
    }

\end{algorithme}
