\begin{algorithme}


    \begin{enregistrement}{TableDeCodage}
        \champEnregistrement{nbElements}{\naturel}
        \champEnregistrement{table}{\tableau{0..255}{de}{Donnees}}
    \end{enregistrement}

    \begin{enregistrement}{Donnees}
        \champEnregistrement{octet}{Octet}
        \champEnregistrement{code}{CodeBinaire}
    \end{enregistrement}


    \fonction{tableDeCodage}
        {}{TableDeCodage}
        {}{resultat : TableDeCodage}
    {
        \affecter{resultat.nbElements}{0}
        \retourner{resultat}
    }


    \fonction{estVide}
        {table : TableDeCodage}{\booleen}
        {}{}
    {\retourner{table.nbElements = 0}}


    \fonction{elementPresent}
        {table : TableDeCodage, octet : Octet}{\booleen}
        {}{i:\naturelNonNul}
    {
        \pour{i}{1}{table.nbElements}{}{
            \sialors{octet = table.table[i].octet}{
                {\retourner{vrai}}
            }
        }
        \retourner{faux}
    }


    \fonction{codePresent}
        {table : TableDeCodage, code : CodeBinaire}{\booleen}
        {}{i:\naturelNonNul}
    {
        \pour{i}{1}{table.nbElements}{}{
            \sialors{code = table.table[i].code}{
                {\retourner{vrai}}
            }
        }
        \retourner{faux}
    }


    \procedure{ajouterElement}{
        \paramEntree{octet : Octet, code : CodeBInaire}
        \paramEntreeSortie{table : TableDeCodage}}
        {ajouterElement(table, octet) et non(elementPresent(table, octet))}
        {}
    {
        \sialors{non(elementPresent(table, octet))}{
            \affecter{table.nbElements}{table.nbElements + 1}
            \affecter{table.table[nbElements]}{octet}
            \affecter{table.table[nbElements]}{code}
            }
    }

    \fonction{obtenirCodeOctet}
        {table : TableDeCodage, octet : Octet}{CodeBinaire}
        {octetPresent(table, octet)}{i:\naturelNonNul}
    {
        \pour{i}{1}{table.nbElements}{}{
            \sialors{(table.table[i].octet = octet)}{
                {\retourner{table.table.codeBinaire}}
            }
        }
    }


    \fonction{obtenirElementCode}
        {table : TableDeCodage, code : CodeBinaire}{Octet}
        {elementPresent(table, element)}
        {i:\naturelNonNul}
    {
        \pour{i}{1}{table.nbElements}{}{
            \sialors{(table.table[i].code = code)}{
            {\retourner{table.table[i].octet}}
            }
        }
    }


\end{algorithme}
