\begin{algorithme}

    \begin{enregistrement}{TableDeCodage}
        \champEnregistrement{taille}{\naturel}
        \champEnregistrement{element}{Element}
        \champEnregistrement{code}{Code}
    \end{enregistrement}

    \fonction{tableDeCodage}
    {}{TableDeCodage}
    {}{}
    {b $\geq$0}{}
    {\retourner{a/b}}

    \fonction{estVide}
    {table : TableDeCodage}{\booleen}
    {}{}
    {\retourner{table.taille = 0}}

    \fonction{elementPresent}
    {table : TableDeCodage, element : Element}{\booleen}
    {}{}
    \pourChaque{table.element}{table}{
        \sialors{table.cle = element}
        {\retourner{vrai}}
        }
    {\retourner{faux}}

    \fonction{codePresent}
    {table : TableDeCodage, code : CodeBinaire}{\booleen}
    {}{}
    \pourChaque{table.element}{table}{
        \sialors{table.code = code}
        {\retourner{vrai}}
        }
    {\retourner{faux}}

    \procedure{ajouterElement}{
        \paramEntreeSortie{table : TableDeCodage, element : Element}}
    {}{}
    {\sialors{non(elementPresent(table, element))}{
        \affecter{table.element}{element}}}

    \fonction{elementPossedeCode}
    {table : TableDeCodage, element : Element}{\booleen}
    {}{}
    \pourChaque{table.element}{table}{
        \sialors{(table.element = code) et (codePresent(table, table.code))}
        {\retourner{vrai}}
        }
    {\retourner{faux}}

    \procedure{assignerCodeElement}{
        \paramEntree{r : \reel}
        \paramEntreeSortie{table : TableDeCodage, element : Element, code : CodeBinaire}}
    {}{}
    {\sialors{non(codePresent(table, code)) et (elementPresent(table, element))}{
        \pourChaque(table.element){table}
            \sialors{table.element = element}    
                \affecter{table.element}{code}} $à revoir en fonction du type$
    }

    \fonction{obtenirCodeElement}
    {table : TableDeCodage, element : Element}{CodeBinaire}
    {}{}
    \pourChaque{table.element}{table}{
        \sialors{(table.element = element) et (codePresent(table, code))}
        {\retourner{code associé à élément}}
        }
    {\retourner{faux}}

    \fonction{obtenirElementCode}
    {table : TableDeCodage, code : CodeBinaire}{Element}
    {}{}
    \pourChaque{table.element}{table}{
        \sialors{(table.element = element) et (codePresent(table, code))}
        {\retourner{table.element}}
        }
\end{algorithme}
