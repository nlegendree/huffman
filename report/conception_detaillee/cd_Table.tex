\begin{algorithme}


    \begin{enregistrement}{TableDeCodage}
        \champEnregistrement{nbElements}{\naturel}
        \champEnregistrement{table}{\tableau{0..255}{de}{Donnees}}
    \end{enregistrement}

    \begin{enregistrement}{Donnees}
        \champEnregistrement{octet}{Octet}
        \champEnregistrement{code}{CodeBinaire}
    \end{enregistrement}


    \fonction{tableDeCodage}
        {}{TableDeCodage}
        {}{tableDeCodage : TableDeCodage}
    {
        \affecter{\champ{tableDeCodage}{nbElements}}{0}
        \retourner{tableDeCodage}
    }


    \fonction{estVide}
        {tableDeCodage : TableDeCodage}{\booleen}
        {}{}
    {\retourner{\champ{tableDeCodage}{nbElements} = 0}}


    \fonction{elementPresent}
        {tableDeCodage : TableDeCodage, o : Octet}{\booleen}
        {}{i : \naturelNonNul}
    {
        \pour{i}{1}{tableDeCodage.nbElements}{}{
            \sialors{o = \champ{tableDeCodage}{table[i]}.octet}{
                {\retourner{vrai}}
            }
        }
        \retourner{faux}
    }


    \fonction{codePresent}
        {tableDeCodage : TableDeCodage, codeBinaire : CodeBinaire}{\booleen}
        {}{i:\naturelNonNul}
    {
        \pour{i}{1}{tableDeCodage.nbElements}{}{
            \sialors{codeBinaire = \champ{tableDeCodage}{table[i]}.code}{
                {\retourner{vrai}}
            }
        }
        \retourner{faux}
    }


    \procedure{ajouterElement}{
        \paramEntreeSortie{tableDeCodage : \typePointeur{TableDeCodage}}
        \paramEntree{octet : Octet, codeBinaire : CodeBinaire}}
        {non(elementPresent(\pointeur{tableDeCodage}, octet)) et (\champPointeur{tableDeCodage}{nbElements} $\leq$ 255)}
        {}
    {
        \affecter{\champPointeur{tableDeCodage}{nbElements}}{\champPointeur{tableDeCodage}{nbElements} + 1}
        \affecter{\champPointeur{tableDeCodage}{table[\champPointeur{tableDeCodage}{nbElements} - 1].octet}}{octet}
        \affecter{\champPointeur{tableDeCodage}{table[\champPointeur{tableDeCodage}{nbElements} - 1].code}}{codeBinaire}
    }

    \fonction{obtenirCodeElement}
        {tableDeCodage : TableDeCodage, o : Octet}{CodeBinaire}
        {elementPresent(tableDeCodage, o)}
        {i : \naturelNonNul}
    {
        \pour{i}{1}{\champ{tableDeCodage}{nbElements}}{}{
            \sialors{(\champ{tableDeCodage}{table[i]}.octet = o)}{
                {\retourner{\champ{tableDeCodage}{table[i]}.code}}
            }
        }
    }


    \fonction{obtenirElementCode}
        {tableDeCodage : TableDeCodage, codeBinaire : CodeBinaire}{Octet}
        {codePresent(tableDeCodage, codeBinaire)}
        {i : \naturelNonNul}
    {
        \pour{i}{1}{\champ{tableDeCodage}{nbElements}}{}{
            \sialors{(\champ{tableDeCodage}{table[i]}.code = codeBinaire)}{
            {\retourner{\champ{tableDeCodage}{table[i]}.octet}}
            }
        }
    }


\end{algorithme}
