\begin{algorithme}


    \begin{enregistrement}{TableDeCodage}
        \champEnregistrement{nbElements}{\naturel}
        \champEnregistrement{Table}{\tableau{0..255}{de}{Donnees}}
    \end{enregistrement}

    \begin{enregistrement}{Donnees}
        \champEnregistrement{octet}{Octet}
        \champEnregistrement{code}{CodeBinaire}
    \end{enregistrement}


    \fonction{tableDeCodage}
        {}{TableDeCodage}
        {}{resultat : TableDeCodage}
    {
        \affecter{resultat.nbElements}{0}
        \retourner{resultat}
    }


    \fonction{estVide}
        {table : TableDeCodage}{\booleen}
        {}{}
    {\retourner{table.nbElements = 0}}


    \fonction{elementPresent}
        {table : TableDeCodage, octet : Octet}{\booleen}
        {}{}
    {
        \pourChaque{table.Table.elements}{table}{
            \sialors{octet = table.Table.octet}{
                {\retourner{vrai}}
            }
        }
        \retourner{faux}
    }


    \fonction{codePresent}
        {table : TableDeCodage, code : CodeBinaire}{\booleen}
        {}{}
    {
        \pourChaque{table.Table.octet}{table}{
            \sialors{table.Table.code = code}{
            {\retourner{vrai}}
            }
        }
        \retourner{faux}
    }


    \procedure{ajouterElement}{
        \paramEntree{octet : Octet}
        \paramEntreeSortie{table : TableDeCodage}}
        {ajouterElement(table, octet) et non(elementPresent(table, octet))}
        {}
    {
        \sialors{non(elementPresent(table, octet))}{
            \affecter{table.nbElements}{table.nbElements + 1}
            \affecter{table.Table}{octet}
            }
    }


    \fonction{elementPossedeCode}
        {table : TableDeCodage, octet : Octet}{\booleen}
        {elementPresent(table, octet)}
        {}
    {
        \pourChaque{table.Table.octet}{table}{
            \sialors{obtenirCodeElement(table.Table, octet) = table.lesElements.octet}{
            {\retourner{vrai}}
            }
        }
        \retourner{faux}
    }


    \procedure{assignerCodeElement}{
        \paramEntree{octet : Octet, code : CodeBinaire}
        \paramEntreeSortie{table : TableDeCodage}}
        {non(codePresent(table, code)) et elementPresent(table, octet)}
        {}
    {
        \sialors{non(codePresent(table.Table, code)) et (elementPresent(table.Table, octet))}{
            \pourChaque{table.Table.octet}{table}{
                \sialors{table.Table.octet = octet}{
                    {\affecter{table.Table.octet}{code}}
                }
            }
        }
    }


    \fonction{obtenirCodeElement}
        {table : TableDeCodage, octet : Octet}{CodeBinaire}
        {}{}
    {
        \pourChaque{table.Table.octet}{table}{
            \sialors{(table.Table.octet = octet) et (codePresent(table.Table, table.Table.octet))}{
                {\retourner{table.Table.octet}}
            }
        }
    }


    \fonction{obtenirElementCode}
        {table : TableDeCodage, code : CodeBinaire}{Octet}
        {elementPossedeCode(table, octet) et elementPresent(table, octet)}
        {}
    {
        \pourChaque{table.Table.octet}{table}{
            \sialors{(table.Table.octet = octet) et (codePresent(table.Table, code))}{
            {\retourner{table.Table.code}}
            }
        }
    }


\end{algorithme}
